from __future__ import print_function 

import optparse 

import os 

import gzip 

import sys 

import zipfile 

 

from pyi_updater.exceptions import ArchiverError 

from pyi_updater.utils import (get_version_number, 

                               get_terminal_size, 

                               version_string_to_tuple) 

from pyi_updater.version import __version__ 

 

max_help_position = 80 

max_width = get_terminal_size()[0] 

 

fmt = optparse.IndentedHelpFormatter(width=max_width, 

                                     max_help_position=max_help_position) 

usage = ('usage: pyi-archive -n "My App" -v 1.0.1 FILE [FILE...]\n' 

         'Usage: pyi-archive -i gzip -n "My App" -v 1.0.1 FILE [FILE...]') 

kw = { 

    'version': __version__, 

    'usage': usage, 

    'formatter': fmt, 

    'conflict_handler': 'resolve', 

    } 

parser = optparse.OptionParser(**kw) 

parser.add_option('-c', '--archiver', 

                  default='zip', 

                  type='choice', 

                  choices=['zip', 'gzip', 'g', 'z'], 

                  help='Type of archive compression to use') 

 

parser.add_option('-n', '--name', help='Name of update') 

 

parser.add_option('-v', '--version-num', 

                  help="Version # of update. Must have Major.Minor.Patch " 

                  "even if it's 0 eg. 1.1.0") 

parser.add_option('--keep', 

                  action='store_true', 

                  default=False, 

                  help='Do not delete source file') 

 

keep = False 

 

 

def main(my_opts=None): 

    global keep 

    if my_opts is None: 

        opts, args = parser.parse_args(sys.argv) 

        keep = opts.keep 

        version = check_version(opts) 

        name = check_name(opts) 

        archiver = opts.archiver 

    else: 

        version = check_version(my_opts) 

        name = check_name(my_opts) 

        args = my_opts.args 

        archiver = my_opts.archiver 

 

    files = [] 

    bad_files = [] 

    for f in args[1:]: 

        if not os.path.exists(f) or not support_files(f): 

            bad_files.append(f) 

        else: 

            files.append(f) 

    # Used for testing purposes 

    if len(files) < 1: 

        return False 

 

    if archiver == 'zip' or 'z': 

        zip_archive(name, version, files) 

    else: 

        gzip_archive(name, version, files) 

 

    if len(bad_files) > 0: 

        print('Files not found:') 

        for f in bad_files: 

            print(f) 

    # Used for testing purposes 

    return True 

 

 

def check_version(options): 

    try: 

        version = options.version 

    except: 

        raise ArchiverError('You must pass -v option with version # of update', 

                            expected=True) 

 

    try: 

        version = get_version_number(version) 

    except: 

        raise ArchiverError('Can not parse version number', 

                            expected=True) 

 

    try: 

        version_tuple = version_string_to_tuple(version) 

    except: 

        raise ArchiverError('Version not in correct format. eg. "0.0.1"', 

                            expected=True) 

 

    if len(version_tuple) != 3: 

        raise ArchiverError('Version must be MAJOR.MINOR.PATCH', 

                            expected=True) 

    return version 

 

 

def check_name(options): 

    try: 

        name = options.name 

    except: 

        raise ArchiverError('You must pass -n option with name of update') 

        sys.exit(1) 

    return name 

 

 

def gzip_archive(name, version, files): 

    for f in files: 

        basename = os.path.basename(f) 

        file_dir = os.path.dirname(f) 

 

        filename = '{}-{}-{}.gz'.format(os.path.join(file_dir, name), 

                                        basename, version) 

        with gzip.open(filename, 'wb') as gf: 

            with open(f, 'rb') as in_file: 

                gf.write(in_file.read()) 

 

        if keep is False: 

            os.remove(f) 

 

 

def zip_archive(name, version, files): 

    for f in files: 

        basename = os.path.basename(f) 

        file_dir = os.path.dirname(f) 

 

        filename = '{}-{}-{}.zip'.format(os.path.join(file_dir, name), 

                                         basename, version) 

        with zipfile.ZipFile(filename, 'w') as zf: 

            zf.write(f, name) 

        if keep is False: 

            os.remove(f) 

 

 

def support_files(f): 

    if f in ['win', 'mac', 'nix', 'nix64', 'arm']: 

        return True 

    return False 

 

if __name__ == '__main__': 

    main()
